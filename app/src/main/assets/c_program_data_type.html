DATA TYPE

#include stdio.h
int main()
{
//Data Types
int a = 5;
char b = b;
long c = 6;
short d = 7;
float e = 1.000000;
double f = 2.000000000000000;

//Modifier types
short int g = 8;
long int h = 9;
long long i = 11;
unsigned int j = 12;
unsigned long k = 13;
unsigned short l = 14;
unsigned short int m = 15;
unsigned long int n = 16;
unsigned long long o = 17;

printf("Data types\n");
printf("int value: %d, int size: %ld\n", a, sizeof(a));
printf("char value: %c, char size: %ld\n", b, sizeof(b));
printf("long value: %ld, long size: %ld\n", c, sizeof(c));
printf("short value: %d, short size: %ld\n", d, sizeof(d));
printf("float value: %f, float size: %ld\n", e, sizeof(e));
printf("double value: %f, double size: %ld\n", f, sizeof(f));
printf("\nModifier types\n");
printf("short int value: %d, short int size: %ld\n", g, sizeof(g));
printf("long int value: %ld, long int size: %ld\n", h, sizeof(h));
printf("long long value: %lld, long long size: %ld\n", i, sizeof(i));
printf("unsigned int value: %d, unsigned int size: %ld\n", j, sizeof(j));
printf("unsigned long value: %ld, unsigned long size: %ld\n", k, sizeof(k));
printf("unsigned short value: %d, unsigned short size: %ld\n", l, sizeof(l));
printf("unsigned short int value: %d, unsigned short int size: %ld\n", m, sizeof(m));
printf("unsigned long int value: %ld, unsigned long int size: %ld\n", n, sizeof(n));
printf("unsigned long long value: %lld, unsigned long long size: %ld\n", o, sizeof(o));

return 0;
}

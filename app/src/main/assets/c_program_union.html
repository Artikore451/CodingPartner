// Comparison of Structures and Unions
#include <stdio.h>
#include <string.h> 

struct sIDCard {
   char Branch[24];
   char Name[20];
   int RegNo;
   int RollNo;
} ID_s;

union uIDCard {
   char Branch[24];
   char Name[20];
   int RegNo;
   int RollNo;
} ID_u;

void main() {
   //  STRUCTURE
   printf("\nFrom Structure\n**********************");
   
   strcpy( ID_s.Branch, "R&A");
   strcpy(ID_s.Name, "Aakash Paul K");
   ID_s.RegNo = 525;
   ID_s.RollNo = 1;
   
   printf("\nBranch\t: %s", ID_s.Branch);
   printf("\nName\t: %s", ID_s.Name);
   printf("\nReg No\t: %d", ID_s.RegNo);
   printf("\nRoll No\t: %d", ID_s.RollNo);

   //  UNION
   printf("\n\nFrom Union\n**********************");
   
   strcpy( ID_u.Branch, "R&A");
   strcpy(ID_u.Name, "Aakash Paul K");
   ID_u.RegNo = 525;
   ID_u.RollNo = 1;
   
   printf("\nBranch\t: %s", ID_u.Branch);
   printf("\nName\t: %s", ID_u.Name);
   printf("\nReg No\t: %d", ID_u.RegNo);
   printf("\nRoll No\t: %d", ID_u.RollNo);
   
   //  UNION - Storing one by one and printing
   printf("\n\nFrom Union - One by one - read and print");
   printf("\n*********************************************");
   
   strcpy( ID_u.Branch, "R&A");
   printf("\nBranch\t: %s", ID_u.Branch);
   strcpy(ID_u.Name, "Aakash Paul K");
   printf("\nName\t: %s", ID_u.Name);
   ID_u.RegNo = 525;
   printf("\nReg No\t: %d", ID_u.RegNo);
   ID_u.RollNo = 1;
   printf("\nRoll No\t: %d", ID_u.RollNo);
   
   //  SIZE OF STRUCTURE & UNION
   printf("\n\nSize of structure variable ID_s : %ld", sizeof(ID_s));
   printf("\nSize of Union variable ID_u \t: %ld", sizeof(ID_u));
}

File handling-

//#include <cstdlib>    // for the definition of EXIT_FAILURE
#include <iostream>
#include <string>
#include <fstream>
using namespace std;

// Associate stream objects with external file names

int main() {

    //*** Writting to a file ***/
    
    ofstream myfile; // outs is an output stream                
    myfile.open("Test.txt"); // connect outs to file outFile
    myfile << "Writing this to a file.\n";
    myfile << "Writing second line to a file.\n";
    myfile.close();    // close output file stream


    //*** Reading from a file ***/
    string line;
    ifstream myfilein; // myfilein is an input stream                
    myfilein.open("Test.txt"); // connect myfilein to file Input File
    if (myfilein.is_open())
    {
        while (getline(myfilein, line))
        {
            cout << line << '\n';
        }
        myfile.close();    // close output file stream
    }
    return 0;    // successful return
}

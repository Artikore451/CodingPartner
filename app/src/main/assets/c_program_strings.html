/* c program to find Length of string without strlen()*/
#include <stdio.h>
int main()
{
    char str[1000];               // character array to store string  
    int i;

    printf("Enter a string:\n");
    fgets(str, 1000, stdin);      // Read string from stdin, of maximum Length 1000          

    for(i = 0; str[i] != '\0'; i++); // '\0' end of string

    printf("Length of string: %d", i);
    return 0;
}
